23   create table company

26       company_name      varchar(255),
27       city varchar(30)
28   );

31   select * from company;
32   select employe.id,employe.person_name from employe INNER JOIN works on employe.person_name=works.person_name and works.company_name='BigBank';
33   select employe.id,employe.person_name,employe.city from employe INNER JOIN works on employe.person_name=works.person_name and works.company_name='BigBank';
34   select employe.id,employe.person_name,employe.city from employe INNER JOIN works on employe.person_name=works.person_name and works.company_name='BigBank' and works.salary>=10000;
35   select employe.id,employe.person_name from employe INNER JOIN works on employe.person_name = works.person_name INNER JOIN company on works.company_name = company.company_name and employe.city = company.city;
36   select employe.id,employe.person_name from employe INNER JOIN works on employe.person_name=works.person_name and works.company_name!='BigBank';
37   select employe.id,employe.person_name from employe INNER JOIN works on employe.person_name=works.person_name and works.salary=(select max(salary) from works);
